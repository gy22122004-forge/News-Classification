Transformer Architecture — Deep Dive

How Attention Is All You Need enables zero-shot news classification

①
Input Embedding

Token IDs → 1024-dim dense vectors. Positional encoding adds sequence order:

PE(pos, 2i) = sin(pos / 100002i/d)

All tokens processed simultaneously — no sequential bottleneck.

→
②
Multi-Head Self-Attention

Each token attends to every other token in parallel:

Attn(Q,K,V) = softmax(QKᵀ/√d_k)·V

16 heads (BART-large) capture syntax, semantics, co-reference simultaneously.

→
③
Encoder Stack (12×)

Each layer: Multi-Head Attn → Add&Norm → FFN (4096 units, GELU) → Add&Norm.

Outputs 1024-dim contextual representation per token for the full article.

→
④
NLI Classification Head

For each label: "This text is about Technology"

Model scores entailment probability → highest score = top category. No fine-tuning!

Core Formula Attention(Q, K, V) = softmax( Q Kᵀ / √d_k ) · V

Python — PyTorch Bidirectional LSTM · Supervised Training rnn_classifier.py 
import torch
import torch.nn as nn
from torchtext.datasets import AG_NEWS
from torchtext.data.utils import get_tokenizer

# ─── Bidirectional LSTM Classifier ──────────────────────────────────────────
class RNNClassifier(nn.Module):
    """Bidirectional LSTM for news classification.
    Processes tokens sequentially — forward + backward pass.
    Source: bentrevett/pytorch-sentiment-analysis (GitHub)"""

    def __init__(self,
                 vocab_size=95811,  # AG News vocab size
                 embed_dim=128,
                 hidden_dim=256,
                 num_classes=8,       # 8 news categories
                 num_layers=2,
                 dropout=0.3):
        super().__init__()
        # Embedding: token ID → 128-dim learned vector
        self.embedding = nn.Embedding(vocab_size, embed_dim, padding_idx=0)
        # Bidirectional LSTM: processes sequence fwd + bwd
        self.lstm = nn.LSTM(embed_dim, hidden_dim,
                            num_layers=num_layers,
                            batch_first=True,
                            dropout=dropout,
                            bidirectional=True)  # ← key: bidirectional!
        self.dropout = nn.Dropout(dropout)
        # Classifier: hidden_dim*2 (fwd+bwd) → 8 classes
        self.fc = nn.Linear(hidden_dim * 2, num_classes)

    def forward(self, x):
        embedded = self.dropout(self.embedding(x))      # [B, L, E]
        _, (h_n, _) = self.lstm(embedded)               # Process sequentially
        # Concatenate last forward + backward hidden states
        hidden = torch.cat((h_n[-2], h_n[-1]), dim=1)  # [B, H*2]
        return self.fc(self.dropout(hidden))            # [B, 8]

# ─── Training on AG News (120k articles, 4 classes) ─────────────────────────
tokenizer  = get_tokenizer("basic_english")
train_iter = AG_NEWS(split="train")                   # 120,000 labelled articles

model      = RNNClassifier(num_classes=4)
optimizer  = torch.optim.SGD(model.parameters(), lr=5.0, momentum=0.9)
loss_fn    = nn.CrossEntropyLoss()

for epoch in range(10):                               # ~15 min on CPU
    for label, text in train_iter:
        output = model(vocab(tokenizer(text)))
        loss   = loss_fn(output, torch.tensor([label - 1]))
        loss.backward(); optimizer.step()

Python — FastAPI REST Endpoint · Production Backend main.py 
from fastapi import FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from fastapi.staticfiles import StaticFiles
from pydantic import BaseModel
from transformers import pipeline
import time, re, os

app = FastAPI(title="News Classifier API", version="1.0.0")
app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_methods=["*"])

# Load model once at startup (cached in memory ~1.6 GB)
classifier = pipeline("zero-shot-classification",
                      model="facebook/bart-large-mnli")

CATEGORIES = ["Politics", "Sports", "Business", "Technology",
              "Health", "Entertainment", "Science", "World News"]

class ClassifyRequest(BaseModel):
    text: str

@app.post("/classify")
def classify_news(req: ClassifyRequest):
    text = req.text.strip()
    if len(text) < 10:
        raise HTTPException(400, "Text too short")

    start = time.time()
    result = classifier(text[:2000],           # truncate for performance
                        candidate_labels=CATEGORIES,
                        multi_label=False)
    elapsed_ms = (time.time() - start) * 1000

    return {
        "top_category":      result["labels"][0],
        "top_score":         result["scores"][0],
        "categories":        [
            {"label": l, "score": s, "percentage": round(s*100,2)}
            for l, s in zip(result["labels"], result["scores"])
        ],
        "processing_time_ms": round(elapsed_ms, 1),
        "word_count":         len(re.findall(r"\b\w+\b", text)),
        "char_count":         len(text),
    }

# Serve frontend at /  — run: uvicorn main:app --port 8000 --reload
app.mount("/", StaticFiles(directory="frontend", html=True), name="static")

Python — Zero-Shot Pipeline · Minimal Demo pipeline_demo.py 
from transformers import pipeline

# The complete demo in 8 lines of code — no training data needed!
classifier = pipeline(
    "zero-shot-classification",
    model="facebook/bart-large-mnli"
)

texts = [
    "Apple unveiled the M4 chip with an AI neural engine at WWDC.",
    "Manchester City won the UEFA Champions League final 3-1.",
    "The Senate passed a landmark $370B climate bill.",
    "Harvard researchers reversed cellular aging by 30% in mice.",
]

CATEGORIES = ["Politics", "Sports", "Business", "Technology",
              "Health", "Entertainment", "Science", "World News"]

for text in texts:
    result = classifier(text, candidate_labels=CATEGORIES, multi_label=False)
    top    = result["labels"][0]
    score  = result["scores"][0]
    print(f"[{top:<15}] {score:.2%}  —  {text[:55]}...")
